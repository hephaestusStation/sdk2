class Client:
    """Client

    The client contains information for connecting to a running Jaxon service
    It provides functions for creating projects, datasets and running various
    workflows on imported datasets such as labeling, training models. The
    models and the curated datasets can be exported via this client.
    """


[docs]    def __init__(
            self, base_url: str = None, client_id: str = None, client_secret: str = None
    ):
        """Initialize the JaxonClient to access the Jaxon server.

        To successfully create a client a valid base_url must be provided.

        >>> import jaxon_client as jc
        >>> client = jc.Client(base_url="http://localhost:31415", client_id="user_name", client_secret="password")

        Args:
            base_url (str): the base URL to a running Jaxon service.
            client_id (str): the id to authenticate.
            client_secret (str): the secret for the id to authenticate
        """

        self._client_config = ClientConfig(base_url, client_id, client_secret)
        self._project_service = ProjectService(self._client_config)
        self._dataset_service = DatasetService(self._client_config)
        self._classical_model_service = ClassicalModelService(self._client_config)
        self._heuristic_service = HeuristicService(self._client_config)
        self._ensemble_service = EnsembleService(self._client_config)
        self._neural_model_service = NeuralModelService(self._client_config)
        self._job_service = JobService(self._client_config)

[docs]    def train_model(self,
                    model_spec: Union[ClassicalModelSpec, NeuralModelSpec],
                    max_wait_time: int = 0) -> Tuple[Result, Optional[Job]]:

        """Trains a model
        
        Trains a classical or neural model based on model_spec type passed in as an argument.

        Args:
           Union[ClassicalModelSpec, NeuralModelSpec]: The model_spec specifying requried parameters for specific model.
           max_wait_time: optional, max wait time in seconds, if zero the call is treated as non-blocking.
        Returns:
            Result: result object carries the response status and an error-message in case of failure.
            Job: used to monitor the job in case the max_wait_time is set to zero.
        Examples:
            >>> classical_model_spec = ClientUtils.classical_model_spec(name="catboost",
                                                                description="train-a-classical-modal",
                                                                project_id=project.id,
                                                                training_dataset_id=train_ds.id,
                                                                test_dataset_id=test_ds.id,
                                                                algorithm=Classical.Algorithm.CAT_BOOST)
            >>>
            >>> result, model = client.train_model(classical_model_spec, max_wait_time=6000)
            >>> if result.is_failure:
            ...     print("Error: {}".format(result.error_message))
            >>> elif job.is_complete:
            ...     result, model = client.get_classical_model(project.id, job.target_id)
            ...     if result.is_failure:
            ...         print("Error: {}".format(result.error_message))
            >>> else:
            ...     print("Job did not complete..")
            >>>
        """

        if isinstance(model_spec, ClassicalModelSpec):
            return self._train_classical_model(model_spec, max_wait_time)
        elif isinstance(model_spec, NeuralModelSpec):
            return self._train_neural_model(model_spec, max_wait_time)

        error_message = f"""
                             Unknown model spec type {type(model_spec)} 
                             expecting type ClassicalModelSpec or NeuralModelSpec
                             """
        raise TypeError(error_message)
